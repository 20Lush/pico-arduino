237  float measureDutyCyclePercent(uint gpio) {
238  Logger.debug("PicoPWMReader::measureDutyCycle");
239  float result = 0;
240  if (pin_mode==INPUT) {
241  // Only the PWM B pins can be used as inputs.
242  if(pwm_gpio_to_channel(gpio) == PWM_CHAN_B) {
243  // used by interrupt handler
244  pico_pwm_actual_pin = gpio;
245  uint slice_num = pwm_gpio_to_slice_num(gpio);
246 
247  // Mask our slice's IRQ output into the PWM block's single interrupt line,
248  // and register our interrupt handler
249  pwm_clear_irq(slice_num);
250  pwm_set_irq_enabled(slice_num, true);
251  irq_set_exclusive_handler(PWM_IRQ_WRAP, on_pico_pwm_wrap);
252  irq_set_enabled(PWM_IRQ_WRAP, true);
253 
254  // measure input pin
255  uint64_t counter = count(slice_num, PWM_READ_REPEAT);
256 
257  // clean up irq
258  pwm_clear_irq(slice_num);
259  pwm_set_irq_enabled(slice_num, false);
260  irq_remove_handler (PWM_IRQ_WRAP, on_pico_pwm_wrap);
261 
262  // convert to percent
263  result = 100.0 * counter / max_ticks_per_period;
264  } else {
265  Logger.error("measureDutyCycle() only allowed the PWM B pins");
266  }
267  } else {
268  Logger.error("measureDutyCycle() only allowed with pinMode INPUT");
269  }
270  return result;
271  }

326  uint64_t count(uint slice_num, int repeat) {
327  uint64_t sleep_period_us = (period_nano_sec / 1000 ); // 2040
328  pico_pwm_wrap_count = 0;
329 
330  if (Logger.isLogging(PicoLogger::Debug)){
331  char sleep_str[70];
332  sprintf(sleep_str,"sleep_us: %lu", sleep_period_us);
333  Logger.debug(sleep_str);
334  }
335 
336  uint64_t counter = 0;
337  for (int j=0;j<repeat;j++) {
338  // measure n pwm cycles
339  pwm_set_counter (slice_num, 0);
340  pwm_set_enabled(slice_num, true);
341  sleep_us((sleep_period_us )); // sleep microsendonds
342  pwm_set_enabled(slice_num, false);
343 
344  counter += pwm_get_counter(slice_num);
345  }
346  counter += (pico_pwm_wrap_count * PWM_MAX_NUMER);
347  return counter / repeat;
